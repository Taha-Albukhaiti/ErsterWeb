Einlesen von float-Werten in eine Reihung und Ausgabevariationen! 
              

                import java.util.Scanner;
            /**
             * Einlesen von float-Werten in eine Reihung und Ausgabevariationen
             *
             */

            public class ReiheEinAus {

                private float[] reell;

                public ReiheEinAus(int laenge) {
                    // neue Reihung vom Variablentyp float mit der uebergebenen Laenge
                    // erstellen
                    reell = new float[laenge];
                }

                public void einlesen() {
                    Scanner sc = new Scanner(System.in);
                    for (int i = 0; i < reell.length; i++) {
                        System.out.print("Bitte eine Zahl: ");
                        reell[i] = sc.nextFloat();
                    }
                    sc.close();
                }

                public void ausgeben() {
                    // 1. Werte der Reihung untereinander ausgeben
                    for (int i = 0; i < reell.length; i++) {
                        System.out.println(reell[i]);
                    }
                    System.out.println();

                    // Merke die folgende for Schleife seit java 5.0:
                    // alle Werte des Arrays reell werden der Variable f der Reihe nach
                    // zugewiesen.
                    // Sie leistet das gleiche wie die vorige Schleife.
                    // Nur sinnvoll, wenn kein Zugriff auf Index noetig.
                    // Insbesondere nicht einsetzbar in der Methode einlesen
                    for (float f : reell) {
                        System.out.println(f);
                    }
                    System.out.println();

                    // 2. Werte rueckwaerts in einer Zeile ausgeben
                    for (int i = reell.length - 1; i >= 0; i--) {
                        System.out.print(reell[i] + "  ");
                    }
                    System.out.println();

                    // 3. Werte an geradem Index ausgeben
                    for (int i = 0; i < reell.length; i += 2) {
                        System.out.println(reell[i]);
                    }
                }
            }



Dieses Programm demonstriert, wie Zahlen ueber der Tastatur gelesen werden, in ein Array gespeichert werden, und wie das Minimum berechnet und ausgegeben wird. 
             

               import java.util.Scanner;

               /**
                * Dieses Programm demonstriert, wie Zahlen ueber der Tastatur gelesen werden,
                * in ein Array gespeichert werden, und wie das Minimum berechnet und ausgegeben
                * wird.
                *
                */

               public class Minimum {

                   public static void main(String args[]) {
                       int anzahl, zaehler, minimum;
                       Scanner sc = new Scanner(System.in);
                       System.out.println(
                           "Programm zur Ermittlung der kleinsten Zahl in einer nicht leeren Zahlenreihe");
                       System.out
                           .println("Die ganzen Zahlen werden ueber Tastatur eingelesen.");
                       System.out.print(
                           "Bitte Groesse des Feldes eingeben - Groesse muss strikt positiv sein: ");

                       anzahl = sc.nextInt();

                       // Deklarieren und Erzeugen der Reihung
                       int[] zahlenreihe = new int[anzahl];

                       // Einlesen ganzer Zahlen in die Reihung:
                       for (zaehler = 0; zaehler < anzahl; zaehler++) {
                           System.out.print("Zahl " + zaehler + ": ");
                           zahlenreihe[zaehler] = sc.nextInt();
                       }
                       sc.close();
                       // Annnahme, die erste Zahl sei die kleinste
                       // erste Zahl hat Index 0!
                       minimum = zahlenreihe[0];
                       // Vergleich mit allen Zahlen der Reihung
                       // Ist eine Zahl kleiner als der Wert in minimum, dann diese
                       // Zahl in minimum speichern
                       for (zaehler = 1; zaehler < anzahl; zaehler++)
                           if (minimum > zahlenreihe[zaehler])
                               minimum = zahlenreihe[zaehler];
                       System.out.println("Minimum ist " + minimum);
                   }

               }
